class Solution {
    public String reverseVowels(String s) {
        char[] ch = s.toCharArray();
        int i = 0 , l = s.length()-1;
        while ( i < l ) {
            while ( i < l && "aeiouAEIOU".indexOf(ch[i]) == -1 ) i++;
            while ( i < l && "aeiouAEIOU".indexOf(ch[l]) == -1 ) l--;
            char temp = ch[i];
            ch[i] = ch[l];
            ch[l] = temp;
            i++;
            l--;
        }
        return new String(ch);
    }
}

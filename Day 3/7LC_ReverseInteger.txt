class Solution {
    public int reverse(int x) {
        String s = Integer.toString(Math.abs(x));
        String rev = "";
        for (int i = s.length()-1; i >= 0; i--) {
            rev += s.charAt(i);
        }
        int res = Integer.parseInt(rev);
        if (x < 0) {
            return -res;
        }
        return res;
    }
}
